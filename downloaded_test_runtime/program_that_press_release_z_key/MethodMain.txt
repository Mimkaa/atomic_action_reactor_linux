public static void main(String[] args){
    System.out.println("About to press 'z' using java.awt.Robot...");
    try{
        java.awt.Robot r=new java.awt.Robot();
        r.setAutoDelay(50);
        r.delay(500);
        r.keyPress(java.awt.event.KeyEvent.VK_Z);
        r.keyRelease(java.awt.event.KeyEvent.VK_Z);
        System.out.println("Pressed 'z'.");
    }catch(Exception e){
        System.out.println("Failed to press key: "+e);
        e.printStackTrace();
    }

    // Terminate pipeline loop: create .ready without using CreateTextFile
    try{
        java.nio.file.Path p=java.nio.file.Paths.get(".ready");
        if(!java.nio.file.Files.exists(p)){
            java.nio.file.Files.createFile(p);
        }
        System.out.println("Created .ready");
    }catch(Exception e){
        System.out.println("Failed to create .ready: "+e);
        e.printStackTrace();
    }
}
